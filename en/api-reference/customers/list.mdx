---
title: "List customers"
description: "Retrieve a paginated list of customers"
openapi: "/api-reference/openapi.json GET /v1/customers"
---

Returns a paginated list of customers for the authenticated merchant.

## Parameters

<ParamField query="page" type="integer">
  Page number. Default: `1`.
</ParamField>

<ParamField query="pageSize" type="integer">
  Number of items per page. Default: `20`. Maximum: `100`.
</ParamField>

<ParamField query="status" type="string">
  Filter by status: `Active` or `Inactive`.
</ParamField>

<ParamField query="documentType" type="string">
  Filter by document type: `CPF` or `CNPJ`.
</ParamField>

<ParamField query="search" type="string">
  Search by name, email, or document number.
</ParamField>

<ParamField query="externalId" type="string">
  Filter by your external ID.
</ParamField>

<ParamField query="startDate" type="string">
  Start date filter in ISO 8601 format.
</ParamField>

<ParamField query="endDate" type="string">
  End date filter in ISO 8601 format.
</ParamField>
